---
- hosts: ansible
  become: true

  tasks:
  - name: create docker image using war file
    command: docker build -t demo-image:latest .
    args:
      chdir: /opt/docker

  - name: create tag to image
    command: docker tag demo-image k2r2t2/demo-image

  - name: push image on to dockerhub
    command: docker push k2r2t2/demo-image

  - name: remove docker images form ansible server
    command: docker rmi demo-image:latest k2r2t2/demo-image
    ignore_errors: yes
